// ---------------- CONFIGURATION ----------------
int number_of_svgs = 100;   
int master_seed    = 1; 

int CANVAS_W = 900;
int CANVAS_H = 900;
int MARGIN   = 50;

String output_folder = "dataset";


String[] KINDS = {
//  "flow_field", // désactivépour le moment car SVG un peu trop grand
  "subdivision", "subdivision",
  "radial_ca", "radial_ca",
  "sketchy_lsys", "sketchy_lsys",
  "circle_packing"
};

float WARP_SCALE;
float WARP_AMP;
float WARP_THETA;
boolean DO_WARP = true;

HashMap<String, LSystemRule> LS = new HashMap<String, LSystemRule>();

void settings() {
  size(900, 900); 
}

void setup() {
  initLSystems();
  ensureDir(output_folder);
}

void draw() {
  for (int i = 0; i < number_of_svgs; i++) {
    generateAndExport(i);
    println("Generated " + (i+1) + "/" + number_of_svgs);
  }
  exit();
}

void generateAndExport(int index) {
  int seed = hash32(master_seed ^ (index * 6849));
  
  randomSeed(seed);
  noiseSeed(seed);
  
  int pick = abs(hash32(seed + 99991)) % KINDS.length;
  String kind = KINDS[pick];

  
  DO_WARP = random(1) < 0.85; 
  WARP_SCALE = random(0.002, 0.005);
  WARP_AMP   = random(0, 5);
  WARP_THETA = random(TWO_PI);

  beginRecord(SVG, output_folder + "/" + nf(index, 4) + "_" + kind + ".svg");
  
  background(255);
  stroke(0);
  noFill();
  strokeCap(ROUND);
  strokeWeight(1);

  if (kind.equals("flow_field")) {
    drawFlowField();
  } else if (kind.equals("subdivision")) {
    drawRecursiveSubdivision();
  } else if (kind.equals("radial_ca")) {
    drawRadialCA();
  } else if (kind.equals("sketchy_lsys")) {
    drawSketchyLSystem();
  } else if (kind.equals("circle_packing")) {
    drawComplexPacking();
  }

  endRecord();
}


void drawFlowField() {
  int numParticles = int(random(800, 2000));
  int steps = int(random(50, 150));
  float stepLen = random(3, 8);
  float noiseScale = random(0.003, 0.01);
  
  float m = MARGIN * random(0.5, 2.0);
  
  for (int i = 0; i < numParticles; i++) {
    float x = random(m, width - m);
    float y = random(m, height - m);
    
    for (int n = 0; n < steps; n++) {
      float angle = noise(x * noiseScale, y * noiseScale) * TWO_PI * 2.0;
      
      float x2 = x + cos(angle) * stepLen;
      float y2 = y + sin(angle) * stepLen;
      
      if (x2 < m || x2 > width - m || y2 < m || y2 > height - m) break;
      
      if (DO_WARP) wLine(x, y, x2, y2);
      else line(x, y, x2, y2);
      
      x = x2;
      y = y2;
      
      if (random(1) < 0.02) break;
    }
  }
}


void drawRecursiveSubdivision() {
  DO_WARP = random(1) < 0.3;
  
  ArrayList<float[]> rects = new ArrayList<float[]>();
  rects.add(new float[]{MARGIN, MARGIN, width - MARGIN*2, height - MARGIN*2});
  
  int iterations = int(random(15, 60));
  
  for (int i = 0; i < iterations; i++) {
    int idx = int(random(rects.size()));
    float[] r = rects.get(idx);
    float rw = r[2];
    float rh = r[3];
    
    if (rw < 40 || rh < 40) continue;
    
    rects.remove(idx);
    
    float splitRatio = random(0.3, 0.7);
    
    if (random(1) < 0.5) {
      float w1 = rw * splitRatio;
      float w2 = rw - w1;
      rects.add(new float[]{r[0], r[1], w1, rh});
      rects.add(new float[]{r[0] + w1, r[1], w2, rh});
    } else {
      float h1 = rh * splitRatio;
      float h2 = rh - h1;
      rects.add(new float[]{r[0], r[1], rw, h1});
      rects.add(new float[]{r[0], r[1] + h1, rw, h2});
    }
  }
  
  for (float[] r : rects) {
    float x = r[0], y = r[1], w = r[2], h = r[3];
    float cx = x + w/2;
    float cy = y + h/2;
    
    float pad = 5;
    x+=pad; y+=pad; w-=pad*2; h-=pad*2;
    
    int type = int(random(5));
    
    if (type == 0) {
      float spac = random(3, 8);
      for (float k = -h; k < w; k += spac) {
        wLineClip(x + k, y, x + k + h, y + h, x, y, w, h);
      }
    } else if (type == 1) {
      int rings = int(min(w, h) / random(4, 10));
      for (int k = 0; k < rings; k++) {
        float dia = min(w, h) * (k / (float)rings);
        wCircle(cx, cy, dia);
      }
    } else if (type == 2) {
      int count = int(w * h / 50.0);
      for(int k=0; k<count; k++) {
        float px = random(x, x+w);
        float py = random(y, y+h);
        if(DO_WARP) wCircle(px, py, 1);
        else circle(px, py, 1);
      }
    }
    
    wRect(r[0]+pad, r[1]+pad, r[2]-pad*2, r[3]-pad*2);
  }
}

void wRect(float x, float y, float w, float h) {
  wLine(x, y, x+w, y);
  wLine(x+w, y, x+w, y+h);
  wLine(x+w, y+h, x, y+h);
  wLine(x, y+h, x, y);
}

void wLineClip(float x1, float y1, float x2, float y2, float bx, float by, float bw, float bh) {

  float nx1 = constrain(x1, bx, bx+bw);
  float ny1 = constrain(y1, by, by+bh);
  float nx2 = constrain(x2, bx, bx+bw);
  float ny2 = constrain(y2, by, by+bh);
  wLine(nx1, ny1, nx2, ny2);
}


void drawRadialCA() {
  DO_WARP = true;
  WARP_AMP = random(2, 10);
  
  int rule = int(random(255));
  int cols = int(random(100, 200));
  int rows = int(random(100, 250)); 
  
  boolean[] cells = new boolean[cols];
  for(int i=0; i<cols; i++) cells[i] = random(1) < 0.5;
  boolean[] next = new boolean[cols];
  
  float maxRadius = (min(width, height) / 2) - MARGIN;
  float minRadius = maxRadius * 0.15;
  float cx = width/2;
  float cy = height/2;
  
  for (int r = 0; r < rows; r++) {
    float rad = map(r, 0, rows, minRadius, maxRadius);
    float radNext = map(r+1, 0, rows, minRadius, maxRadius);
    
    for (int i = 0; i < cols; i++) {
      if (cells[i]) {
        float theta1 = map(i, 0, cols, 0, TWO_PI);
        float theta2 = map(i+1, 0, cols, 0, TWO_PI);
        

        float x1 = cx + cos(theta1) * rad;
        float y1 = cy + sin(theta1) * rad;
        float x2 = cx + cos(theta2) * rad;
        float y2 = cy + sin(theta2) * rad;
        
        wLine(x1, y1, x2, y2);
        
        if (random(1) < 0.3) {
           float x3 = cx + cos(theta1) * radNext;
           float y3 = cy + sin(theta1) * radNext;
           wLine(x1, y1, x3, y3);
        }
      }
    }
    
    for (int i = 0; i < cols; i++) {
      int left = cells[(i - 1 + cols) % cols] ? 1 : 0;
      int me   = cells[i] ? 1 : 0;
      int right= cells[(i + 1) % cols] ? 1 : 0;
      int idx = (left << 2) | (me << 1) | right;
      next[i] = ((rule >> idx) & 1) == 1;
    }
    boolean[] temp = cells; cells = next; next = temp;
  }
}

ArrayList<float[]> interpretToSegmentsSketch(String instr, float baseAngle, float jitter, float step) {
  ArrayList<float[]> segs = new ArrayList<float[]>();
  float x = 0, y = 0, h = 0; 

  ArrayDeque<float[]> stack = new ArrayDeque<float[]>();

  for (int i = 0; i < instr.length(); i++) {
    char c = instr.charAt(i);

    if (c == 'F' || c == 'G' || c == 'A' || c == 'B') { 
      float hNow = h + random(-jitter, jitter);
      float st = step * random(0.85, 1.15); 

      float x2 = x + cos(hNow) * st;
      float y2 = y + sin(hNow) * st;

      segs.add(new float[]{x, y, x2, y2});
      x = x2; y = y2;

    } else if (c == '+') {
      h += baseAngle + random(-0.15, 0.15); 
    } else if (c == '-') {
      h -= baseAngle + random(-0.15, 0.15);
    } else if (c == '[') {
      stack.push(new float[]{x, y, h});
    } else if (c == ']') {
      if (!stack.isEmpty()) {
        float[] st = stack.pop();
        x = st[0]; y = st[1]; h = st[2];
      }
    }
  }
  return segs;
}



void drawSketchyLSystem() {
  String[] keys = LS.keySet().toArray(new String[0]);
  String key = keys[int(random(keys.length))];
  LSystemRule r = LS.get(key);

  int iters = int(random(r.minIter, r.maxIter + 1));

  boolean isOrganic = key.equals("sto_tree") || key.equals("bush") || key.equals("weed") || key.equals("algae") || key.equals("plant");

  float baseAngle = radians(r.angleDeg);
  if (isOrganic) baseAngle += random(-0.1, 0.1);

  float globalRot = random(TWO_PI);
  float flipX = (random(1) < 0.5) ? -1 : 1;
  float flipY = (random(1) < 0.5) ? -1 : 1;
  float stepBase = random(7.0, 14.0);

  String instr = applyRules(r.axiom, r.rules, iters);

  ArrayList<float[]> cleanSegs = interpretToSegmentsSketch(instr, baseAngle, 0, stepBase, false);
  float[] bounds = getBounds(cleanSegs);

  float bw = bounds[2] - bounds[0];
  float bh = bounds[3] - bounds[1];
  float avail = min(width, height) - MARGIN * 2;
  float sc = avail / max(bw, bh);

  float ox = MARGIN + (avail - bw * sc) / 2 - bounds[0] * sc;
  float oy = MARGIN + (avail - bh * sc) / 2 - bounds[1] * sc;

  int passes = int(random(1, 1));

  for (int p = 0; p < passes; p++) {
    float visualJitter = 0;
    
    ArrayList<float[]> segs = interpretToSegmentsSketch(instr, baseAngle, visualJitter, stepBase + random(-1, 1), isOrganic);

    for (float[] s : segs) {
      if (random(1) < 0.10) continue; 

      float x1 = s[0] * sc + ox;
      float y1 = s[1] * sc + oy;
      float x2 = s[2] * sc + ox;
      float y2 = s[3] * sc + oy;

      PVector a = transformPoint(x1, y1, width/2, height/2, globalRot, flipX, flipY);
      PVector b = transformPoint(x2, y2, width/2, height/2, globalRot, flipX, flipY);

      a.x += random(-1, 1); a.y += random(-1, 1);
      b.x += random(-1, 1); b.y += random(-1, 1);

      wLine(a.x, a.y, b.x, b.y);
    }
  }
}

ArrayList<float[]> interpretToSegmentsSketch(String instr, float baseAngle, float visualJitter, float step, boolean isOrganic) {
  ArrayList<float[]> segs = new ArrayList<float[]>();
  float x = 0, y = 0, h = 0; 
  ArrayDeque<float[]> stack = new ArrayDeque<float[]>();

  for (int i = 0; i < instr.length(); i++) {
    char c = instr.charAt(i);

    if (c == 'F' || c == 'G' || c == 'A' || c == 'B') { 
      float drawAngle = h + random(-visualJitter, visualJitter);
      
      float st = step * (isOrganic ? random(0.8, 1.2) : random(0.95, 1.05)); 

      float x2 = x + cos(drawAngle) * st;
      float y2 = y + sin(drawAngle) * st;

      segs.add(new float[]{x, y, x2, y2});
      x = x2; y = y2;

    } else if (c == '+') {

      float structuralNoise = isOrganic ? random(-0.1, 0.1) : 0;
      h += baseAngle + structuralNoise;
      
    } else if (c == '-') {
      float structuralNoise = isOrganic ? random(-0.1, 0.1) : 0;
      h -= baseAngle + structuralNoise;
      
    } else if (c == '[') {
      stack.push(new float[]{x, y, h});
    } else if (c == ']') {
      if (!stack.isEmpty()) {
        float[] st = stack.pop();
        x = st[0]; y = st[1]; h = st[2];
      }
    }
  }
  return segs;
}
PVector transformPoint(float x, float y, float cx, float cy, float rot, float fx, float fy) {
  float dx = x - cx;
  float dy = y - cy;

  dx *= fx;
  dy *= fy;

  float cr = cos(rot);
  float sr = sin(rot);
  float rx = dx * cr - dy * sr;
  float ry = dx * sr + dy * cr;

  return new PVector(cx + rx, cy + ry);
}


ArrayList<float[]> interpretToSegments(String instr, float angleStep, float jitter) {
  ArrayList<float[]> segs = new ArrayList<float[]>();
  float x=0, y=0, h=0, step=10;
  ArrayDeque<float[]> stack = new ArrayDeque<float[]>();
  
  for (int i=0; i<instr.length(); i++) {
    char c = instr.charAt(i);
    if (c=='F' || c=='G') {
      float hNow = h + random(-jitter, jitter);
      float x2 = x + cos(hNow)*step;
      float y2 = y + sin(hNow)*step;
      segs.add(new float[]{x, y, x2, y2});
      x=x2; y=y2;
    } else if (c=='+') h += angleStep;
    else if (c=='-') h -= angleStep;
    else if (c=='[') stack.push(new float[]{x, y, h});
    else if (c==']') {
      float[] st = stack.pop();
      x=st[0]; y=st[1]; h=st[2];
    }
  }
  return segs;
}

float[] getBounds(ArrayList<float[]> segs) {
  float minX=1e9, minY=1e9, maxX=-1e9, maxY=-1e9;
  for(float[] s : segs) {
    minX=min(minX, s[0]); minY=min(minY, s[1]);
    maxX=max(maxX, s[0]); maxY=max(maxY, s[1]);
  }
  return new float[]{minX, minY, maxX, maxY};
}


void drawComplexPacking() {
  ArrayList<PVector> circles = new ArrayList<PVector>(); 

  int attempts = int(random(10000, 40000));     
  int maxCircles = int(random(500, 2000));      
  
  float minR = random(2, 5);
  float maxR = random(30, 100);                 

  float padding = 0.5; 

  boolean useEllipse = random(1) < 0.2; 

  for (int i = 0; i < attempts && circles.size() < maxCircles; i++) {
    float x = random(MARGIN, width - MARGIN);
    float y = random(MARGIN, height - MARGIN);
    float r = random(minR, maxR);

    boolean ok = true;
    for (PVector c : circles) {
      float d = dist(x, y, c.x, c.y);
      if (d < c.z + r + padding) { 
        ok = false; 
        break; 
      }
    }

    if (!ok) continue;

    circles.add(new PVector(x, y, r));

    
    if (useEllipse && r > 25) {
      float ratio = random(0.6, 0.95);
      float ang = random(TWO_PI);
      
      pushMatrix();
      translate(x, y);
      rotate(ang);
      ellipse(0, 0, r*2, r*2*ratio);
      
      int rings = int(r / 4);
      for (int k = 1; k < rings; k++) {
        float rr = map(k, 0, rings, r, 0);
        ellipse(0, 0, rr*2, rr*2*ratio);
      }
      popMatrix();
      
    } else {
      circle(x, y, r * 2);

      float texType = random(1);
      
      if (texType < 0.3) {
        circle(x, y, max(1, r * 0.15));
        
      } else if (texType < 0.6) {
        float a = random(TWO_PI);
        line(x - cos(a)*r, y - sin(a)*r, x + cos(a)*r, y + sin(a)*r);
        
      } else if (texType < 0.9) {
        float step = random(3, 8);
        for (float rr = r - step; rr > 1; rr -= step) {
          circle(x, y, rr * 2);
        }
      }
    }
  }
}



PVector warpPoint(float x, float y) {
  if (!DO_WARP) return new PVector(x, y);
  
  float n = noise(x * WARP_SCALE, y * WARP_SCALE);
  float angle = map(n, 0, 1, 0, TWO_PI * 2) + WARP_THETA;
  float off = map(noise(x*WARP_SCALE + 500, y*WARP_SCALE + 500), 0, 1, 0, WARP_AMP);
  
  return new PVector(x + cos(angle)*off, y + sin(angle)*off);
}

void wLine(float x1, float y1, float x2, float y2) {
  PVector p1 = warpPoint(x1, y1);
  PVector p2 = warpPoint(x2, y2);
  line(p1.x, p1.y, p2.x, p2.y);
}

void wCircle(float x, float y, float d) {
  PVector c = warpPoint(x, y);
  float n = noise(x*0.01, y*0.01);
  float dWarp = d * (0.9 + n*0.2); 
  circle(c.x, c.y, dWarp);
}

void wEllipse(float x, float y, float w, float h, float rot) {
  PVector c = warpPoint(x, y);
  pushMatrix();
  translate(c.x, c.y);
  rotate(rot);
  ellipse(0, 0, w, h);
  popMatrix();
}

int hash32(int x) {
  x ^= (x << 13);
  x ^= (x >> 17);
  x ^= (x << 5);
  return x;
}

void ensureDir(String name) {
  File f = new File(sketchPath(name));
  if (!f.exists()) f.mkdirs();
}
class LSystemRule {
  String name;
  String axiom;
  HashMap<Character, String[]> rules; 
  float angleDeg;
  int minIter, maxIter;

  LSystemRule(String name, String axiom, HashMap<Character, String[]> rules,
              float angleDeg, int minIter, int maxIter) {
    this.name = name;
    this.axiom = axiom;
    this.rules = rules;
    this.angleDeg = angleDeg;
    this.minIter = minIter;
    this.maxIter = maxIter;
  }
}

void initLSystems() {

  tree.put('X', new String[]{
    "F[+X]F[-X]+X",      // Classique
    "F[-X]F[+X]-X",      // Inversé
    "F[+X]-X",           // Branche manquante
    "F[-X]+X",           // Autre branche manquante
    "F[+X]F[-X]F[+X]-X"  // Dense
  });
  LS.put("sto_tree", new LSystemRule("StoTree", "X", tree, 22, 5, 7));

  HashMap<Character, String[]> bush = new HashMap<Character, String[]>();
  bush.put('F', new String[]{"FF-[-F+F+F]+[+F-F-F]"}); 
  LS.put("bush", new LSystemRule("Bush", "F", bush, 20, 3, 5));

  HashMap<Character, String[]> algae = new HashMap<Character, String[]>();
  algae.put('A', new String[]{"[+A]B[-A]B", "[+A]B", "B[-A]B"}); 
  algae.put('B', new String[]{"AA", "A"});
  LS.put("algae", new LSystemRule("Algae", "A", algae, 35, 6, 9));

  HashMap<Character, String[]> crystal = new HashMap<Character, String[]>();
  crystal.put('F', new String[]{"F+F-F-F+F", "F-F+F+F-F"}); 
  LS.put("crystal", new LSystemRule("Crystal", "F-F-F-F", crystal, 90, 3, 5));

  HashMap<Character, String[]> dragon = new HashMap<Character, String[]>();
  dragon.put('X', new String[]{"X+YF+", "X+YF"}); 
  dragon.put('Y', new String[]{"-FX-Y", "-FX-Y"});
  LS.put("dragon", new LSystemRule("Dragon", "FX", dragon, 90, 9, 13));

  HashMap<Character, String[]> terdragon = new HashMap<Character, String[]>();
  terdragon.put('F', new String[]{"F+F-F"});
  LS.put("terdragon", new LSystemRule("Terdragon", "F", terdragon, 120, 5, 8));

  HashMap<Character, String[]> moore = new HashMap<Character, String[]>();
  moore.put('L', new String[]{"-RF+LFL+FR-"});
  moore.put('R', new String[]{"+LF-RFR-FL+"});

  HashMap<Character, String[]> minkowski = new HashMap<Character, String[]>();
  minkowski.put('F', new String[]{"F+F-F-FF+F+F-F"});

  HashMap<Character, String[]> cross = new HashMap<Character, String[]>();
  cross.put('F', new String[]{"F+F-F-F-F+F+F+F-F"});

  HashMap<Character, String[]> rings = new HashMap<Character, String[]>();
  rings.put('F', new String[]{"F+F-F-FFF+F+F-F"});
  LS.put("rings", new LSystemRule("Rings", "F+F+F+F", rings, 90, 2, 3));

  HashMap<Character, String[]> board = new HashMap<Character, String[]>();
  board.put('F', new String[]{"FF+F+F+F+FF"});
  LS.put("board", new LSystemRule("Board", "F+F+F+F", board, 90, 3, 4));

  HashMap<Character, String[]> weed = new HashMap<Character, String[]>();
  weed.put('F', new String[]{"F[+F]F[-F]F"});
  LS.put("weed", new LSystemRule("Weed", "F", weed, 25.7, 4, 6));
}

String applyRules(String axiom, HashMap<Character, String[]> rules, int iterations) {
  String current = axiom;
  for (int i = 0; i < iterations; i++) {
    StringBuilder next = new StringBuilder();
    for (int j = 0; j < current.length(); j++) {
      char c = current.charAt(j);
      
      if (rules.containsKey(c)) {
        String[] options = rules.get(c);
        String choice = options[int(random(options.length))];
        next.append(choice);
      } else {
        next.append(c);
      }
    }
    current = next.toString();
  }
  return current;
}
